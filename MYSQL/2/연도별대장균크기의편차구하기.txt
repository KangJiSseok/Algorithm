SELECT YEAR(A.DIFFERENTIATION_DATE) AS YEAR,
    ABS(A.SIZE_OF_COLONY - C.MAX_SIZE) AS YEAR_DEV,
    A.ID
FROM ECOLI_DATA A
LEFT JOIN (
    SELECT YEAR(B.DIFFERENTIATION_DATE) AS YEAR,
        MAX(B.SIZE_OF_COLONY) AS MAX_SIZE
    FROM ECOLI_DATA B
    GROUP BY YEAR(B.DIFFERENTIATION_DATE)
) C
ON YEAR(A.DIFFERENTIATION_DATE) = C.YEAR
ORDER BY YEAR ASC, YEAR_DEV ASC
